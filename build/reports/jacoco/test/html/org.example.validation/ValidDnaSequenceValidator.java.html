package org.example.validation;

import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.regex.Pattern;

public class ValidDnaSequenceValidator implements ConstraintValidator<ValidDnaSequence, String[]> {

    // Compilamos el patrón una sola vez para eficiencia
    private static final Pattern DNA_PATTERN = Pattern.compile("^[ATCG]+$");
    private static final int MIN_SIZE = 4; // 4x4 es el mínimo para una secuencia de 4

    @Override
    public boolean isValid(String[] dna, ConstraintValidatorContext context) {
        // 1. Validar que no sea nulo o vacío
        if (dna == null || dna.length == 0) {
            // @NotNull y @NotEmpty se encargan de esto, pero es buena práctica
            return false;
        }

        final int N = dna.length;

        // 2. Validar tamaño mínimo
        if (N < MIN_SIZE) {
            return false;
        }

        // 3. Validar que sea NxN y caracteres válidos
        for (String row : dna) {
            // 3a. Validar que la fila no sea nula y tenga longitud N
            if (row == null || row.length() != N) {
                return false;
            }
            // 3b. Validar caracteres (A, T, C, G)
            if (!DNA_PATTERN.matcher(row).matches()) {
                return false;
            }
        }

        return true; // Si pasa todas las validaciones
    }
}
